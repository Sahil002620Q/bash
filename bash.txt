#detailed
ls -l
file permission | number of links | owner name | owner group | file size | last modification time in month date time format | file/dir name

ls -a
hidden files in .file and .folder format

ls -lh
convert file size in kb mb or gb in file size row rest full format is same as ls -l

ls -t
sort dir and files by modification time

ls -r 
reverse the output can be used with other arg

ls -R
show dir with sub dir recursively

ls -S
sort by size

ls -1
one file/dir per line

ls -d */
only dir with no content
